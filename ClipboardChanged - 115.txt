            state.fileCount++
            
            FileGetSize, fileSize, %filePath%
            if (fileSize != "") {

            if (fileTime > state.lastModified) {

        }
    }

FormatSize(size) {
    if (size >= 1073741824)
        Return Round(size / 1073741824, 1) . " GB"
    else if (size >= 1048576)
        Return Round(size / 1048576, 1) . " MB"
    else if (size >= 1024)
        Return Round(size / 1024, 1) . " KB"
    else if (size > 0)
        Return size . " B"
    else
        Return "0 B"
}
